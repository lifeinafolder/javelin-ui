Carousel

Carousel plugin allows you to create a carousel of UI controls; images for example.

Required Includes

<script src="plugins/carousel/Carousel.js" type="text/javascript"></script>
<link rel="stylesheet" type="text/css" href="plugins/carousel/Carousel.css"/>

Required Markup

<div class="carousel">
  <!-- Carousel items -->
  <div data-sigil="carousel-item active"><img src="http://twitter.github.com/bootstrap/assets/img/bootstrap-mdo-sfmoma-01.jpg"/></div>
  <div data-sigil="carousel-item"><img src="http://twitter.github.com/bootstrap/assets/img/bootstrap-mdo-sfmoma-02.jpg"></div>
  <div data-sigil="carousel-item"><img src="http://twitter.github.com/bootstrap/assets/img/bootstrap-mdo-sfmoma-03.jpg"/></div>
  <!-- Carousel nav -->
  <div data-sigil="carousel-control prev">&lsaquo;</div>
  <div data-sigil="carousel-control next">&rsaquo;</div>
</div>

Code

JX.Stratcom.mergeData(0, {});
JX.behavior('carousel-effect', function(config, statics) {
  var selector = config.selector;
  delete config.selector;
  var carousel = new JX.Carousel(selector, config);
});

JX.initBehaviors({
  'carousel-effect' : [{selector:'.carousel',interval:2000, pause:'hover'}],
});
