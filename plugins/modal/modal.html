Modal

Modal plugin allows you to draw user attention to your system messages.

Required Includes

<script src="plugins/keyboard.js" type="text/javascript"></script>
<script src="javelin/src/ext/fx/FX.js" type="text/javascript"></script>
<script src="plugins/modal/Modal.js" type="text/javascript"></script>
<link rel="stylesheet" type="text/css" href="plugins/modal/modal.css"/>

Code

JX.Stratcom.mergeData(0, {});
JX.initBehaviors({
  'show-modal' : [{source : "#my-modal"}]
});

Actionable Modal

Actionable Modal plugin allows you to lock your UI to enforce a user action.

Required Includes

<script src="plugins/keyboard.js" type="text/javascript"></script>
<script src="javelin/src/ext/fx/FX.js" type="text/javascript"></script>
<script src="plugins/modal/Modal.js" type="text/javascript"></script>
<link rel="stylesheet" type="text/css" href="plugins/modal/modal.css"/>

Required Markup

<div id="my-actionable-modal">
  <div data-sigil="modal-header"></div>
  <div data-sigil="modal-body"></div>
  <div data-sigil="modal-footer"></div>
</div>

Code

JX.Stratcom.mergeData(0, {});

JX.initBehaviors({
  'actionable-modal' : [{source : "#my-actionable-modal", action: function(cbk){
    console.log('Do something with this modal:', this);
    cbk && cbk(true);
  }}]
});
