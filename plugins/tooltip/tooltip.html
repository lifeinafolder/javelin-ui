Tooltip

Tooltip plugin allows you to place a custom tooltip on DOM elements.

Required Includes

<script src="../memoize.js" type="text/javascript"></script>
<script src="plugins/tooltip/tooltip.js" type="text/javascript"></script>
<link rel="stylesheet" type="text/css" href="plugins/tooltip/tooltip.css"/>

Required Markup

<p data-meta="0_0" data-sigil="tooltip" title="test title">Dom element must have:
  <ol>
    <li> A data-sigil attribute of 'tooltip'
    <ol> A title attribute for tooltip text
    <ol> A data-meta attribute indicating position of config options in mergeData map
  </ol>
</p>

Code

JX.Stratcom.mergeData(0, [
  {
    trigger: 'hover'  // Config options for Example 1
  },
  {
    trigger:'focus',  // Config options for Example 2
    position:'right'
  }
]);

JX.initBehaviors({
  'show-tooltip' : []
});
