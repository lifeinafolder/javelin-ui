Dropdown

Dropdown plugin allows you to create navigational dropdowns. This is not a typical dropdown. Its a UI control semantically equivalent to a navigational control.

Required Includes

<script src="plugins/toggle/Toggle.js" type="text/javascript"></script>
<script src="plugins/dropdown/Dropdown.js" type="text/javascript"></script>
<link rel="stylesheet" type="text/css" href="plugins/Dropdown/dropdown.css"/>

Required Markup

<div id="dropdown">
  <div data-sigil="dropdown-handle">-</div>
  <ul data-sigil="dropdown-menu">
    <li data-value="one" data-sigil="dropdown-item">Option 1</li>
    <li data-value="two" data-sigil="dropdown-item">Option 2</li>
    <li data-value="three" data-sigil="dropdown-item">Option 3</li>
  </ul>
</div>

Code

JX.Stratcom.mergeData(0, {});

JX.initBehaviors({
  'show-dropdown' : [{
    selector:'#dropdown',
    onItemClick:function(response){
      // Callback function invoked when an item is clicked.
      console.log('Response:', response);
    }
  }]
});
